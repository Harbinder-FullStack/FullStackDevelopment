1. CLI (Command Line Interface)

Input: Arguments, stdin
Output: Console (stdout)

// app.js
const args = process.argv.slice(2);
console.log("Arguments:", args);

process.stdin.on("data", (data) => {
  console.log("You typed:", data.toString().trim());
});

Run: node app.js hello world

2. Browser (HTTP Requests)

Input: HTTP requests
Output: HTML/JSON response

const http = require("http");

http.createServer((req, res) => {
  res.writeHead(200, {"Content-Type": "text/plain"});
  res.end("Hello from Node.js!");
}).listen(3000);

Visit: http://localhost:3000

3. Frontend Apps (React, Angular, etc.)

Input: API requests (fetch/axios)
Output: JSON responses

// Express server
const express = require("express");
const app = express();
app.use(express.json());

app.post("/api/data", (req, res) => {
  res.json({ received: req.body });
});
app.listen(4000, () => console.log("API running on 4000"));

4. File System

Input: File read
Output: File write

const fs = require("fs");
fs.writeFileSync("output.txt", "Hello File!");
console.log(fs.readFileSync("output.txt", "utf8"));

5. Databases

Input: Query requests
Output: Query results

const mongoose = require("mongoose");
mongoose.connect("mongodb://localhost:27017/test");

const User = mongoose.model("User", { name: String });
const user = new User({ name: "John" });
user.save().then(() => console.log("User saved"));

6. API Tools (Postman, curl)

Input: HTTP requests
Output: JSON responses

// Express API
app.get("/api/hello", (req, res) => {
  res.json({ msg: "Hello API!" });
});

Test with curl: curl http://localhost:4000/api/hello

7. Background / Scheduled Jobs

Input: Schedule trigger
Output: Logs, file, DB update

const cron = require("node-cron");
cron.schedule("* * * * *", () => console.log("Runs every minute"));
